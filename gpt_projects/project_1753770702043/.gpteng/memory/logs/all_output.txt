```python
import sqlite3
from models import Todo

def get_db_connection():
    conn = sqlite3.connect('database.db')
    conn.row_factory = sqlite3.Row
    return conn

def init_db():
    conn = get_db_connection()
    conn.execute('''
        CREATE TABLE IF NOT EXISTS todos (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT NOT NULL,
            completed BOOLEAN NOT NULL CHECK (completed IN (0, 1))
        )
    ''')
    conn.commit()
    conn.close()

def add_task(title: str) -> Todo:
    conn = get_db_connection()
    cursor = conn.cursor()
    cursor.execute('INSERT INTO todos (title, completed) VALUES (?, ?)', (title, False))
    conn.commit()
    task_id = cursor.lastrowid
    conn.close()
    return Todo(id=task_id, title=title, completed=False)

def get_tasks() -> list:
    conn = get_db_connection()
    tasks = conn.execute('SELECT * FROM todos').fetchall()
    conn.close()
    return [Todo(id=row['id'], title=row['title'], completed=bool(row['completed'])) for row in tasks]

def complete_task(task_id: int):
    conn = get_db_connection()
    conn.execute('UPDATE todos SET completed = ? WHERE id = ?', (True, task_id))
    conn.commit()
    conn.close()

def delete_task(task_id: int):
    conn = get_db_connection()
    conn.execute('DELETE FROM todos WHERE id = ?', (task_id,))
    conn.commit()
    conn.close()
```

routes.py
```python
from flask import jsonify, request
from database import add_task, get_tasks, complete_task, delete_task, init_db

def init_app(app):
    init_db()

    @app.route('/tasks', methods=['GET'])
    def list_tasks():
        tasks = get_tasks()
        return jsonify([task.__dict__ for task in tasks])

    @app.route('/tasks', methods=['POST'])
    def create_task():
        data = request.get_json()
        task = add_task(data['title'])
        return jsonify(task.__dict__), 201

    @app.route('/tasks/<int:task_id>/complete', methods=['POST'])
    def mark_task_complete(task_id):
        complete_task(task_id)
        return '', 204

    @app.route('/tasks/<int:task_id>', methods=['DELETE'])
    def remove_task(task_id):
        delete_task(task_id)
        return '', 204
```

app.js
```javascript
document.addEventListener('DOMContentLoaded', () => {
    const taskList = document.getElementById('task-list');
    const newTaskInput = document.getElementById('new-task');
    const addTaskButton = document.getElementById('add-task');

    const fetchTasks = async () => {
        const response = await fetch('/tasks');
        const tasks = await response.json();
        taskList.innerHTML = '';
        tasks.forEach(task => {
            const taskItem = document.createElement('li');
            taskItem.textContent = task.title;
            taskItem.className = task.completed ? 'completed' : '';
            const completeButton = document.createElement('button');
            completeButton.textContent = 'Complete';
            completeButton.onclick = () => completeTask(task.id);
            const deleteButton = document.createElement('button');
            deleteButton.textContent = 'Delete';
            deleteButton.onclick = () => deleteTask(task.id);
            taskItem.appendChild(completeButton);
            taskItem.appendChild(deleteButton);
            taskList.appendChild(taskItem);
        });
    };

    const addTask = async () => {
        const title = newTaskInput.value.trim();
        if (title) {
            await fetch('/tasks', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ title })
            });
            newTaskInput.value = '';
            fetchTasks();
        }
    };

    const completeTask = async (taskId) => {
        await fetch(`/tasks/${taskId}/complete`, { method: 'POST' });
        fetchTasks();
    };

    const deleteTask = async (taskId) => {
        await fetch(`/tasks/${taskId}`, { method: 'DELETE' });
        fetchTasks();
    };

    addTaskButton.onclick = addTask;
    fetchTasks();
});
```
