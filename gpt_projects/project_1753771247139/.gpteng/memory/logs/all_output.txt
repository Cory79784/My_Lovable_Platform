#### expense_tracker.py
```python
from typing import List, Dict
from datetime import date
from expense import Expense
from category import Category

class ExpenseTracker:
    """Manages expenses and categories, and generates reports."""
    
    def __init__(self):
        self.expenses: List[Expense] = []
        self.categories: Dict[str, Category] = {}

    def add_category(self, category: Category):
        """Adds a new category."""
        self.categories[category.name] = category

    def add_expense(self, expense: Expense):
        """Adds a new expense."""
        if expense.category not in self.categories:
            raise ValueError(f"Category '{expense.category}' does not exist.")
        self.expenses.append(expense)

    def get_expenses_by_month(self, year: int, month: int) -> List[Expense]:
        """Returns expenses for a specific month and year."""
        return [expense for expense in self.expenses if expense.date.year == year and expense.date.month == month]
```

#### report_generator.py
```python
from typing import List, Dict
from expense import Expense

class ReportGenerator:
    """Generates monthly reports for expenses."""
    
    def generate_monthly_report(self, expenses: List[Expense]) -> Dict[str, float]:
        """Generates a report summarizing expenses by category."""
        report = {}
        for expense in expenses:
            if expense.category in report:
                report[expense.category] += expense.amount
            else:
                report[expense.category] = expense.amount
        return report
```

#### main.py
```python
from datetime import date
from expense import Expense
from category import Category
from expense_tracker import ExpenseTracker
from report_generator import ReportGenerator

def main():
    tracker = ExpenseTracker()
    report_gen = ReportGenerator()

    # Add categories
    tracker.add_category(Category(name="Food", description="Expenses for food and dining"))
    tracker.add_category(Category(name="Transport", description="Expenses for transportation"))

    # Add expenses
    tracker.add_expense(Expense(amount=50.0, category="Food", date=date(2023, 10, 1), description="Groceries"))
    tracker.add_expense(Expense(amount=20.0, category="Transport", date=date(2023, 10, 2), description="Bus fare"))

    # Generate report for October 2023
    expenses_october = tracker.get_expenses_by_month(2023, 10)
    report = report_gen.generate_monthly_report(expenses_october)

    # Print report
    print("Monthly Report for October 2023:")
    for category, total in report.items():
        print(f"{category}: ${total:.2f}")

if __name__ == "__main__":
    main()
```
