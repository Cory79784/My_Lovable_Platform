#### weather_service.py
```python
import requests
from models import WeatherData, ForecastData

API_KEY = 'your_openweathermap_api_key'
BASE_URL = 'http://api.openweathermap.org/data/2.5/'

def get_current_weather(city: str) -> WeatherData:
    response = requests.get(f"{BASE_URL}weather?q={city}&appid={API_KEY}")
    data = response.json()
    return WeatherData.from_api(data)

def get_forecast(city: str) -> ForecastData:
    response = requests.get(f"{BASE_URL}forecast?q={city}&appid={API_KEY}")
    data = response.json()
    return ForecastData.from_api(data)
```

#### models.py
```python
from dataclasses import dataclass
from typing import List

@dataclass
class WeatherData:
    temperature: float
    description: str
    icon: str

    @staticmethod
    def from_api(data: dict) -> 'WeatherData':
        return WeatherData(
            temperature=data['main']['temp'],
            description=data['weather'][0]['description'],
            icon=data['weather'][0]['icon']
        )

@dataclass
class ForecastData:
    forecast: List[WeatherData]

    @staticmethod
    def from_api(data: dict) -> 'ForecastData':
        forecast_list = [
            WeatherData.from_api(item) for item in data['list']
        ]
        return ForecastData(forecast=forecast_list)
```

#### src/CurrentWeather.js
```javascript
import React, { useEffect, useState } from 'react';
import axios from 'axios';
import WeatherCard from './WeatherCard';

function CurrentWeather({ city }) {
  const [weather, setWeather] = useState(null);

  useEffect(() => {
    axios.get(`/api/weather/current?city=${city}`)
      .then(response => setWeather(response.data))
      .catch(error => console.error(error));
  }, [city]);

  return (
    <div>
      <h2>Current Weather</h2>
      {weather && <WeatherCard weather={weather} />}
    </div>
  );
}

export default CurrentWeather;
```

```javascript
import React, { useEffect, useState } from 'react';
import axios from 'axios';
import WeatherCard from './WeatherCard';

function Forecast({ city }) {
  const [forecast, setForecast] = useState([]);

  useEffect(() => {
    axios.get(`/api/weather/forecast?city=${city}`)
      .then(response => setForecast(response.data.forecast))
      .catch(error => console.error(error));
  }, [city]);

  return (
    <div>
      <h2>5-Day Forecast</h2>
      <div className="forecast">
        {forecast.map((weather, index) => (
          <WeatherCard key={index} weather={weather} />
        ))}
      </div>
    </div>
  );
}

export default Forecast;
```
